#include <stdio.h>
#include <string.h>
#define minFinder(a,b) a<b ? a:b

int main() {
    int n =0;
    int m = 0;
    int cntB=0;
    int cntW =0;
    int min =65;
    char board[50][50];
    
    scanf("%d %d",&n,&m);
    for(int i=0;i<n;i++){
        scanf("%s",board[i]);
    }

    for(int i=0;i<n-7;i++){
        for(int j=0;j<m-7;j++){
            cntB = 0;
            cntW=0;
            for(int a=i;a<i+8;a++){
                for(int b=j;b<j+8;b++){
                    if((a+b) % 2 == 0){
                        if(board[a][b] == 'B'){
                            cntW++;
                        }
                        else{
                            cntB++;
                        }
                    }
                    else{
                            if(board[a][b] =='B'){
                                cntB++;
                            }
                            else{
                                cntW++;
                            }
                        }
                    }
                }
            min = minFinder(min, cntB);
            min = minFinder(min, cntW);
            }
        }
    printf("%d\n",min);
    
    return 0;
}
