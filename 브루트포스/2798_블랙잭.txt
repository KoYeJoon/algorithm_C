#include <stdio.h>

int main() {
    int cardNum = 0;
    int cardSum = 0;
    
    scanf("%d %d",&cardNum, &cardSum);
    
    int arr[100] = {0};
    
    for(int i=0; i<cardNum ; i++){
        scanf("%d",&arr[i]);
    }
    

    int max = 0;
    int temp=0;
    int a=0, b=0, c=0;
    for(int i =0 ;i<cardNum-2 ; i++){
        a = arr[i];
        for (int j=i+1;j<cardNum-1;j++) {
            b = arr[j];
            for(int k=j+1; k<cardNum;k++){
                c = arr[k];
                temp = a+b+c;
                if(temp <= cardSum){
                    if(max < temp){
                        max = temp;
                    }
                }
                
            }
        }
    }
    
    printf("%d", max);
    return 0;
}
