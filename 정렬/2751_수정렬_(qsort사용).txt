#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int compare(void *first, void *second){
    if(*(int*)first > *(int*)second)
        return 1;
    else if(*(int*)first < *(int*)second)
        return -1;
    else
        return 0;
}

int main() {
    int num = 0;
    int numArr[1000000] ={0};
    scanf("%d",&num);
    
    for(int i=0;i<num;i++){
        scanf("%d",&numArr[i]);
    }
    
    qsort(numArr, num, sizeof(int), compare);
    for(int i=0;i<num;i++){
        printf("%d\n", numArr[i]);
    }
    return 0;
}
