#include <stdio.h>

int main() {
    int num = 0;
    
    int temp = 0;
    int min = 0;
    

    scanf("%d", &num);
    
    for(int i =0; i<num; i++){
        int numArr[8] = {0,};
        int sum = 0;
        temp = i;
        for(int j=0;j<8;j++){
            if(temp>9){
                numArr[j] = temp%10;
                temp = temp/10;
            }else{
                numArr[j]=temp;
                break;
            }
        }
        for(int j=0;j<8;j++){
            sum = sum + numArr[j];
        }
        
        sum = sum + i;
        
        if(sum == num){
            min=i;
            break;
        }
        
    }
    
    printf("%d", min);
    return 0;
}
